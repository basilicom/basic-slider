Default CSS

.basic-slider {
    overflow: hidden;
    position: relative;
}

Default JavaScript

$(function(){
    $('#slider').basicSlider();

    // calling a method
    $('#slider').basicSlider('method', arguments);
}

Options

width:Number - default: 600
height:Number - default: 300
duration:Number - default: 500
easing:String - default: 'swing'
onStart:Function - default: null
onComplete:Function - default: null
navigationContainer:jQueryElement - default: null; you can use a class selector for multiple navigation's

Public Methods

next()
prev()
goToSlide(int) e.g. 1-3
goToIndex(int) e.g. 0-2

CSS for Navigation

.basic-slider {
    overflow: hidden;
    position: relative;
}

div.navigation {
    text-align: center;
    width: 590px;
    padding: 5px;
}

div.navigation a {
    display:inline-block;
    zoom:1; /* triggers hasLayout in ie for inline-block elements */
    width:10px;
    height: 10px;
    background-color: #cccccc;
    line-height: 10px;
    margin: 2px;
    border-radius: 5px;
}

div.navigation a.active {
    background-color: #009999;
}

JavaScript for Navigation

$(function(){
    $('#slider').basicSlider({
        navigationContainer:$('div.navigation')
    });
}
